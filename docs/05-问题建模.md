## 1. 线性规划 (LP)

### 1.1 问题描述

最简单的优化问题，目标函数和约束都是线性的。

```
max cᵀx
s.t. Ax ≤ b
     x ≥ 0
```

### 1.2 代码实现

```python
"""
线性规划示例: 最大化利润
某工厂生产两种产品，每种产品需要劳动力、原材料。
给定资源限制，求最大利润的生产方案。
"""

import gurobipy as gp
from gurobipy import GRB

def solve_linear_programming():
    # ===== 问题数据 =====
    products = ["P1", "P2"]
    profit = {"P1": 30, "P2": 40}  # 单位利润
    resources = ["labor", "material"]  # 资源类型
    
    # 资源消耗矩阵 (产品 x 资源)
    consumption = {
        ("P1", "labor"): 1,     ("P1", "material"): 2,
        ("P2", "labor"): 2,    ("P2", "material"): 1,
    }
    
    # 资源可用量
    capacity = {"labor": 100, "material": 80}
    
    # ===== 创建模型 =====
    model = gp.Model("linear_programming")
    
    # ===== 决策变量 =====
    # x[i] = 产品i的生产数量
    x = model.addVars(products, vtype=GRB.CONTINUOUS, lb=0, name="x")
    
    # ===== 目标函数 =====
    # 最大化总利润
    model.setObjective(
        gp.quicksum(profit[i] * x[i] for i in products),
        GRB.MAXIMIZE
    )
    
    # ===== 约束条件 =====
    # 每种资源的消耗不超过可用量
    for r in resources:
        model.addConstr(
            gp.quicksum(consumption[i, r] * x[i] for i in products) <= capacity[r],
            name=f"capacity_{r}"
        )
    
    # ===== 求解 =====
    model.optimize()
    
    # ===== 结果 =====
    if model.status == GRB.OPTIMAL:
        print(f"最优利润: {model.objVal}")
        print("生产方案:")
        for i in products:
            print(f"  {i}: {x[i].x}")
    
    return model

if __name__ == "__main__":
    solve_linear_programming()
```

## 2. 整数规划 (IP / MIP)

### 2.1 混合整数规划 (MIP)

```python
"""
混合整数规划示例: 工厂选址
决定开设哪些工厂，使总成本最小。
开设成本为固定成本，加上运营成本。
"""

import gurobipy as gp
from gurobipy import GRB

def solve_facility_location():
    # ===== 问题数据 =====
    potential_facilities = ["F1", "F2", "F3"]
    customers = ["C1", "C2", "C3", "C4"]
    
    # 开设成本 (固定成本)
    fixed_cost = {"F1": 1000, "F2": 800, "F3": 1200}
    
    # 运营成本 (单位成本)
    unit_cost = {"F1": 5, "F2": 6, "F3": 4}
    
    # 客户需求量
    demand = {"C1": 50, "C2": 80, "C3": 60, "C4": 70}
    
    # ===== 创建模型 =====
    model = gp.Model("facility_location_mip")
    
    # ===== 决策变量 =====
    # y[j] = 1 如果开设设施j
    y = model.addVars(potential_facilities, vtype=GRB.BINARY, name="y")
    
    # x[j] = 设施j为客户服务的数量
    x = model.addVars(customers, potential_facilities, 
                      vtype=GRB.CONTINUOUS, lb=0, name="x")
    
    # ===== 目标函数 =====
    # 最小化总成本 = 开设成本 + 运营成本
    model.setObjective(
        gp.quicksum(fixed_cost[j] * y[j] for j in potential_facilities) +
        gp.quicksum(unit_cost[j] * x[i, j] for i in customers for j in potential_facilities),
        GRB.MINIMIZE
    )
    
    # ===== 约束条件 =====
    # 满足所有客户需求
    for i in customers:
        model.addConstr(
            gp.quicksum(x[i, j] for j in potential_facilities) >= demand[i],
            name=f"demand_{i}"
        )
    
    # 只能从已开设设施供货
    for i in customers:
        for j in potential_facilities:
            model.addConstr(x[i, j] <= demand[i] * y[j], name=f"link_{i}_{j}")
    
    # ===== 求解 =====
    model.optimize()
    
    # ===== 结果 =====
    if model.status == GRB.OPTIMAL:
        print(f"最优总成本: {model.objVal}")
        print("开设的设施:")
        for j in potential_facilities:
            if y[j].x > 0.5:
                print(f"  {j}: 成本 {fixed_cost[j]}")
        print("\n供货方案:")
        for i in customers:
            for j in potential_facilities:
                if x[i, j].x > 0.1:
                    print(f"  {i} ← {j}: {x[i, j].x}")
    
    return model
```

## 3. 网络流问题

### 3.1 最小成本流

```python
"""
最小成本流问题: 物资配送
从多个仓库向多个客户配送物资，找到最低运输成本方案。
"""

import gurobipy as gp
from gurobipy import GRB

def solve_min_cost_flow():
    # ===== 问题数据 =====
    warehouses = ["W1", "W2"]
    customers = ["C1", "C2", "C3"]
    
    # 仓库供应量
    supply = {"W1": 100, "W2": 150}
    
    # 客户需求量
    demand = {"C1": 60, "C2": 80, "C3": 90}
    
    # 单位运输成本
    cost = {
        ("W1", "C1"): 4, ("W1", "C2"): 6, ("W1", "C3"): 9,
        ("W2", "C1"): 5, ("W2", "C2"): 3, ("W2", "C3"): 7,
    }
    
    # ===== 创建模型 =====
    model = gp.Model("min_cost_flow")
    
    # ===== 决策变量 =====
    # flow[i,j] = 从仓库i运往客户j的物资数量
    flow = model.addVars(warehouses, customers, 
                         vtype=GRB.CONTINUOUS, lb=0, name="flow")
    
    # ===== 目标函数 =====
    # 最小化总运输成本
    model.setObjective(
        gp.quicksum(cost[i, j] * flow[i, j] 
                    for i in warehouses for j in customers),
        GRB.MINIMIZE
    )
    
    # ===== 约束条件 =====
    # 仓库发货量不超过供应量
    for i in warehouses:
        model.addConstr(
            gp.quicksum(flow[i, j] for j in customers) <= supply[i],
            name=f"supply_{i}"
        )
    
    # 客户收到需求量
    for j in customers:
        model.addConstr(
            gp.quicksum(flow[i, j] for i in warehouses) >= demand[j],
            name=f"demand_{j}"
        )
    
    # ===== 求解 =====
    model.optimize()
    
    # ===== 结果 =====
    if model.status == GRB.OPTIMAL:
        print(f"最低运输成本: {model.objVal}")
        print("配送方案:")
        for i in warehouses:
            for j in customers:
                if flow[i, j].x > 0:
                    print(f"  {i} → {j}: {flow[i, j].x} (成本: {cost[i,j]*flow[i,j].x})")
    
    return model
```

## 4. 组合优化问题

### 4.1 集合覆盖问题

```python
"""
集合覆盖问题: 应急设施选址
选择最少的设施位置，使其覆盖所有需求点。
"""

import gurobipy as gp
from gurobipy import GRB

def solve_set_covering():
    # ===== 问题数据 =====
    facilities = ["F1", "F2", "F3", "F4", "F5"]
    demand_points = ["D1", "D2", "D3", "D4", "D5", "D6"]
    
    # 设施建设成本
    cost = {"F1": 10, "F2": 15, "F3": 8, "F4": 12, "F5": 9}
    
    # 覆盖关系 (设施能覆盖哪些需求点)
    covers = {
        "F1": ["D1", "D2", "D3"],
        "F2": ["D2", "D3", "D4"],
        "F3": ["D3", "D4", "D5"],
        "F4": ["D4", "D5", "D6"],
        "F5": ["D1", "D5", "D6"],
    }
    
    # ===== 创建模型 =====
    model = gp.Model("set_covering")
    
    # ===== 决策变量 =====
    # y[j] = 1 如果选择设施j
    y = model.addVars(facilities, vtype=GRB.BINARY, name="y")
    
    # ===== 目标函数 =====
    # 最小化总建设成本
    model.setObjective(
        gp.quicksum(cost[j] * y[j] for j in facilities),
        GRB.MINIMIZE
    )
    
    # ===== 约束条件 =====
    # 每个需求点至少被一个设施覆盖
    for d in demand_points:
        model.addConstr(
            gp.quicksum(y[j] for j in facilities if d in covers[j]) >= 1,
            name=f"cover_{d}"
        )
    
    # ===== 求解 =====
    model.optimize()
    
    # ===== 结果 =====
    if model.status == GRB.OPTIMAL:
        print(f"最低总成本: {model.objVal}")
        print("选择的设施:")
        for j in facilities:
            if y[j].x > 0.5:
                print(f"  {j}: 覆盖 {covers[j]}")
    
    return model
```

### 4.2 集合分割问题

```python
"""
集合分割问题: 车辆路径规划简化版
将需求点分割成互不重叠的组，每组由一辆车服务。
"""

import gurobipy as gp
from gurobipy import GRB

def solve_set_partitioning():
    # ===== 问题数据 =====
    clients = ["A", "B", "C", "D"]
    vehicles = 2  # 可用车辆数
    
    # 车辆容量
    capacity = 3
    
    # 客户需求
    demand = {"A": 1, "B": 2, "C": 1, "D": 2}
    
    # 预定义的路线及其成本
    routes = [
        {"clients": ["A"], "cost": 10},
        {"clients": ["B"], "cost": 12},
        {"clients": ["C"], "cost": 11},
        {"clients": ["D"], "cost": 13},
        {"clients": ["A", "B"], "cost": 18},
        {"clients": ["A", "C"], "cost": 17},
        {"clients": ["B", "D"], "cost": 20},
        {"clients": ["C", "D"], "cost": 19},
        {"clients": ["A", "B", "C"], "cost": 25},
        {"clients": ["B", "C", "D"], "cost": 27},
    ]
    
    # ===== 创建模型 =====
    model = gp.Model("set_partitioning")
    
    # ===== 决策变量 =====
    # x[k] = 1 如果选择路线k
    x = model.addVars(range(len(routes)), vtype=GRB.BINARY, name="x")
    
    # ===== 目标函数 =====
    # 最小化总成本
    model.setObjective(
        gp.quicksum(routes[k]["cost"] * x[k] for k in range(len(routes))),
        GRB.MINIMIZE
    )
    
    # ===== 约束条件 =====
    # 每个客户恰好被服务一次
    for c in clients:
        model.addConstr(
            gp.quicksum(x[k] for k in range(len(routes)) if c in routes[k]["clients"]) == 1,
            name=f"serve_{c}"
        )
    
    # 使用不超过指定数量的车辆
    model.addConstr(
        gp.quicksum(x[k] for k in range(len(routes))) <= vehicles,
        name="vehicle_limit"
    )
    
    # ===== 求解 =====
    model.optimize()
    
    # ===== 结果 =====
    if model.status == GRB.OPTIMAL:
        print(f"最低总成本: {model.objVal}")
        print("选择的路线:")
        for k in range(len(routes)):
            if x[k].x > 0.5:
                print(f"  路线 {k+1}: {routes[k]['clients']} (成本: {routes[k]['cost']})")
    
    return model
```

## 5. 调度问题

### 5.1 排班问题

```python
"""
排班问题: 员工排班
确定每天需要的员工数量，满足需求的同时最小化成本。
"""

import gurobipy as gp
from gurobipy import GRB

def solve_staff_scheduling():
    # ===== 问题数据 =====
    days = ["Mon", "Tue", "Wed", "Thu", "Fri", "Sat", "Sun"]
    
    # 每天最低员工需求
    demand = {"Mon": 10, "Tue": 15, "Wed": 12, "Thu": 14, 
              "Fri": 18, "Sat": 8, "Sun": 6}
    
    # 员工工作5天休息2天
    WORK_DAYS = 5
    REST_DAYS = 2
    
    # 员工成本
    daily_cost = 100
    overtime_cost = 150
    
    # ===== 创建模型 =====
    model = gp.Model("staff_scheduling")
    
    # ===== 决策变量 =====
    # x[d] = 在d天工作的员工数量
    x = model.addVars(days, vtype=GRB.INTEGER, lb=0, name="x")
    
    # y = 加班员工数量
    y = model.addVar(vtype=GRB.INTEGER, lb=0, name="y")
    
    # ===== 目标函数 =====
    # 最小化总成本
    model.setObjective(
        daily_cost * gp.quicksum(x[d] for d in days) + 
        overtime_cost * y,
        GRB.MINIMIZE
    )
    
    # ===== 约束条件 =====
    # 满足每天需求 (基础员工 + 加班)
    for d in days:
        model.addConstr(x[d] + y >= demand[d], name=f"demand_{d}")
    
    # ===== 求解 =====
    model.optimize()
    
    # ===== 结果 =====
    if model.status == GRB.OPTIMAL:
        print(f"最低总成本: {model.objVal}")
        print("每日员工安排:")
        for d in days:
            base = int(x[d].x)
            over = int(y.x) if d in ["Fri", "Sat"] else 0
            print(f"  {d}: 基础 {base} + 加班 {over} = {base + over} (需求: {demand[d]})")
    
    return model
```

## 6. 装箱问题

### 6.1 一维装箱

```python
"""
一维装箱问题 (Bin Packing)
将物品装入最少数量的箱子。
"""

import gurobipy as gp
from gurobipy import GRB

def solve_bin_packing():
    # ===== 问题数据 =====
    items = [f"item_{i}" for i in range(10)]
    item_weights = [3, 5, 2, 8, 4, 6, 1, 7, 3, 5]
    bin_capacity = 10
    
    # ===== 创建模型 =====
    model = gp.Model("bin_packing")
    
    # 大M值 (箱子数量上界)
    n_bins = len(items)
    
    # ===== 决策变量 =====
    # x[i,k] = 1 如果物品i装入箱子k
    x = model.addVars(items, range(n_bins), vtype=GRB.BINARY, name="x")
    
    # y[k] = 1 如果箱子k被使用
    y = model.addVars(range(n_bins), vtype=GRB.BINARY, name="y")
    
    # ===== 目标函数 =====
    # 最小化使用的箱子数量
    model.setObjective(
        gp.quicksum(y[k] for k in range(n_bins)),
        GRB.MINIMIZE
    )
    
    # ===== 约束条件 =====
    # 每个物品必须装入一个箱子
    for i in items:
        idx = items.index(i)
        model.addConstr(
            gp.quicksum(x[i, k] for k in range(n_bins)) == 1,
            name=f"assign_{i}"
        )
    
    # 箱子容量约束
    for k in range(n_bins):
        model.addConstr(
            gp.quicksum(items.index(i) * item_weights[items.index(i)] * x[i, k] 
                        for i in items) <= bin_capacity * y[k],
            name=f"capacity_{k}"
        )
    
    # 箱子按顺序使用 (对称性破缺)
    for k in range(n_bins - 1):
        model.addConstr(y[k] >= y[k + 1], name=f"ordering_{k}")
    
    # ===== 求解 =====
    model.optimize()
    
    # ===== 结果 =====
    if model.status == GRB.OPTIMAL:
        print(f"最少箱子数量: {model.objVal}")
        print("装箱方案:")
        for k in range(n_bins):
            bin_items = [i for i in items if x[i, k].x > 0.5]
            if bin_items:
                weight = sum(item_weights[items.index(i)] for i in bin_items)
                print(f"  箱子 {k+1}: {bin_items} (总重量: {weight}/{bin_capacity})")
    
    return model
```

## 7. 建模技巧

### 7.1 线性化技巧

#### 乘积的线性化

```python
# 如果 z = x * y 且 x, y 是二进制变量
# 线性化为:
# z <= x
# z <= y
# z >= x + y - 1

model.addConstr(z <= x)
model.addConstr(z <= y)
model.addConstr(z >= x + y - 1)
```

#### 蕴含约束

```python
# if x == 1 then y == 0
# 线性化为:
# y <= 1 - x

model.addConstr(y <= 1 - x)
```

#### 绝对值约束

```python
# |expr| <= M
# 等价于:
# expr <= M
# -expr <= M

model.addConstr(expr <= M)
model.addConstr(-expr <= M)
```
