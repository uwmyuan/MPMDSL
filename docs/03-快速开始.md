## 2. 第一个模型：0-1背包问题

### 2.1 问题描述

```
给定 n 个物品，每个物品有价值 v[i] 和重量 w[i]。
背包容量为 W。
选择物品装入背包，使得总价值最大且总重量不超过 W。

数学模型:
max sum(v[i] * x[i])
s.t. sum(w[i] * x[i]) <= W
     x[i] in {0, 1} for all i
```

### 2.2 Python实现

创建文件 `knapsack.py`:

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
0-1 背包问题 - ORMDSL 示例
"""

import gurobipy as gp
from gurobipy import GRB

def solve_knapsack():
    """求解0-1背包问题"""
    
    # ===== 问题数据 =====
    items = ["item1", "item2", "item3", "item4", "item5"]
    profits = {"item1": 10, "item2": 20, "item3": 15, "item4": 25, "item5": 30}
    weights = {"item1": 2, "item2": 4, "item3": 3, "item4": 5, "item5": 6}
    capacity = 10
    
    # ===== 创建模型 =====
    model = gp.Model("knapsack")
    
    # ===== 决策变量 =====
    # x[i] = 1 如果物品i被选中，否则为0
    x = model.addVars(items, vtype=GRB.BINARY, name="x")
    
    # ===== 目标函数 =====
    # 最大化总利润
    model.setObjective(
        gp.quicksum(profits[i] * x[i] for i in items),
        GRB.MAXIMIZE
    )
    
    # ===== 约束条件 =====
    # 总重量不超过背包容量
    model.addConstr(
        gp.quicksum(weights[i] * x[i] for i in items) <= capacity,
        name="weight_capacity"
    )
    
    # ===== 求解 =====
    model.optimize()
    
    # ===== 输出结果 =====
    print("=" * 50)
    print("0-1 背包问题求解结果")
    print("=" * 50)
    
    if model.status == GRB.OPTIMAL:
        print(f"\n最优目标值: {model.objVal}")
        print(f"求解时间: {model.runtime:.4f} 秒")
        print("\n选中的物品:")
        for i in items:
            if x[i].x > 0.5:
                print(f"  ✓ {i}: 利润={profits[i]}, 重量={weights[i]}")
        
        total_weight = sum(weights[i] * x[i].x for i in items)
        print(f"\n总重量: {total_weight} / {capacity}")
    else:
        print(f"求解失败，状态: {model.status}")
    
    return model

if __name__ == "__main__":
    solve_knapsack()
```


## 3. 第二个模型：设施选址问题

### 3.1 问题描述

```
在 m 个候选设施中选址开设 P 个设施，
为 n 个客户提供服务，使总加权距离最小。

数学模型:
min sum(d[i][j] * demand[i] * x[i][j])
s.t. sum(y[j] for j in J) = P            # 开设P个设施
     sum(x[i][j] for j in J) = 1        # 每个客户恰好分配到一个设施
     x[i][j] <= y[j]                     # 只能分配到开放的设施
     x[i][j] in {0, 1}, y[j] in {0, 1}
```

### 3.2 Python实现

创建文件 `facility_location.py`:

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
P-中位设施选址问题 - ORMDSL 示例
"""

import gurobipy as gp
from gurobipy import GRB

def solve_facility_location():
    """求解P-中位设施选址问题"""
    
    # ===== 问题数据 =====
    customers = ["c1", "c2", "c3", "c4", "c5"]
    facilities = ["f1", "f2", "f3", "f4"]
    P = 2  # 开设设施数量
    
    demand = {"c1": 10, "c2": 15, "c3": 8, "c4": 12, "c5": 5}
    
    distance = {
        ("c1", "f1"): 10, ("c1", "f2"): 15, ("c1", "f3"): 20, ("c1", "f4"): 25,
        ("c2", "f1"): 15, ("c2", "f2"): 10, ("c2", "f3"): 18, ("c2", "f4"): 22,
        ("c3", "f1"): 20, ("c3", "f2"): 18, ("c3", "f3"): 8,  ("c3", "f4"): 16,
        ("c4", "f1"): 25, ("c4", "f2"): 22, ("c4", "f3"): 16, ("c4", "f4"): 10,
        ("c5", "f1"): 18, ("c5", "f2"): 14, ("c5", "f3"): 12, ("c5", "f4"): 15,
    }
    
    # ===== 创建模型 =====
    model = gp.Model("facility_location")
    
    # ===== 决策变量 =====
    # y[j] = 1 如果设施j被开设
    y = model.addVars(facilities, vtype=GRB.BINARY, name="y")
    
    # x[i][j] = 1 如果客户i分配给设施j
    x = model.addVars(customers, facilities, vtype=GRB.BINARY, name="x")
    
    # ===== 目标函数 =====
    # 最小化总加权距离
    model.setObjective(
        gp.quicksum(distance[i, j] * demand[i] * x[i, j] 
                    for i in customers for j in facilities),
        GRB.MINIMIZE
    )
    
    # ===== 约束条件 =====
    # 1. 开设恰好P个设施
    model.addConstr(
        gp.quicksum(y[j] for j in facilities) == P,
        name="num_facilities"
    )
    
    # 2. 每个客户恰好分配到一个设施
    for i in customers:
        model.addConstr(
            gp.quicksum(x[i, j] for j in facilities) == 1,
            name=f"assign_{i}"
        )
    
    # 3. 只能分配到开放的设施
    for i in customers:
        for j in facilities:
            model.addConstr(x[i, j] <= y[j], name=f"link_{i}_{j}")
    
    # ===== 求解 =====
    model.optimize()
    
    # ===== 输出结果 =====
    print("=" * 60)
    print("P-中位设施选址问题求解结果")
    print("=" * 60)
    
    if model.status == GRB.OPTIMAL:
        print(f"\n最优目标值: {model.objVal}")
        print(f"求解时间: {model.runtime:.4f} 秒")
        
        print("\n开设的设施:")
        for j in facilities:
            if y[j].x > 0.5:
                print(f"  ✓ {j}")
        
        print("\n客户分配方案:")
        for i in customers:
            for j in facilities:
                if x[i, j].x > 0.5:
                    print(f"  {i} → {j} (距离: {distance[i, j]})")
    else:
        print(f"求解失败，状态: {model.status}")
    
    return model

if __name__ == "__main__":
    solve_facility_location()
```

### 3.3 运行

```bash
python facility_location.py
```

## 4. 第三个模型：旅行商问题(TSP)

### 4.1 问题描述

```
给定 n 个城市及其之间的距离矩阵，
找到访问所有城市恰好一次并返回起点的最短路径。

数学模型:
min sum(d[i][j] * x[i][j]) 
s.t. sum(x[i][j] for j) = 1           # 每个城市恰好离开一次
     sum(x[i][j] for i) = 1           # 每个城市恰好到达一次
     u[i] - u[j] + n * x[i][j] <= n-1  # MTZ子回路消除约束
     x[i][j] in {0, 1}
```

### 4.2 Python实现

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
旅行商问题 (TSP) - ORMDSL 示例
"""

import gurobipy as gp
from gurobipy import GRB

def solve_tsp(n_cities=5):
    """求解旅行商问题"""
    
    import random
    random.seed(42)
    
    cities = list(range(n_cities))
    
    # 生成随机距离矩阵
    distance = {}
    for i in cities:
        for j in cities:
            if i != j:
                distance[i, j] = random.randint(1, 100)
    
    # ===== 创建模型 =====
    model = gp.Model("tsp")
    
    # ===== 决策变量 =====
    # x[i][j] = 1 如果从城市i前往城市j
    x = model.addVars(cities, cities, vtype=GRB.BINARY, name="x")
    
    # u[i] 用于子回路消除 (MTZ formulation)
    u = model.addVars(cities, vtype=GRB.CONTINUOUS, lb=0, ub=n_cities, name="u")
    
    # ===== 目标函数 =====
    model.setObjective(
        gp.quicksum(distance[i, j] * x[i, j] 
                    for i in cities for j in cities if i != j),
        GRB.MINIMIZE
    )
    
    # ===== 约束条件 =====
    # 每个城市恰好离开一次
    for i in cities:
        model.addConstr(
            gp.quicksum(x[i, j] for j in cities if i != j) == 1,
            name=f"leave_{i}"
        )
    
    # 每个城市恰好到达一次
    for j in cities:
        model.addConstr(
            gp.quicksum(x[i, j] for i in cities if i != j) == 1,
            name=f"arrive_{j}"
        )
    
    # 子回路消除约束 (MTZ)
    for i in cities:
        for j in cities:
            if i != j and i != 0 and j != 0:
                model.addConstr(
                    u[i] - u[j] + n_cities * x[i, j] <= n_cities - 1,
                    name=f"subtour_{i}_{j}"
                )
    
    # ===== 求解 =====
    model.optimize()
    
    # ===== 输出结果 =====
    print("=" * 50)
    print(f"旅行商问题求解结果 (n={n_cities})")
    print("=" * 50)
    
    if model.status == GRB.OPTIMAL:
        print(f"\n最短路径长度: {model.objVal}")
        print(f"求解时间: {model.runtime:.4f} 秒")
        
        # 提取路径
        path = [0]
        current = 0
        for _ in range(n_cities - 1):
            for j in cities:
                if j not in path and x[current, j].x > 0.5:
                    path.append(j)
                    current = j
                    break
        path.append(0)  # 返回起点
        
        print(f"\n最优路径: {' → '.join(map(str, path))}")
    else:
        print(f"求解失败，状态: {model.status}")
    
    return model

if __name__ == "__main__":
    solve_tsp(5)
```

## 7. 常见问题

### Q: 求解结果显示"最优"但解不符合预期？

A: 检查约束是否正确定义，确保约束逻辑与问题一致。

### Q: 求解时间很长？

A: 尝试：
- 设置时间限制 `model.Params.TimeLimit = 3600`
- 设置最优性间隙 `model.Params.MIPGap = 0.001`
- 使用启发式求解 `model.Params.Heuristics = 0.5`

### Q: 如何处理大规模问题？

A: 考虑：
- 分解问题为子问题
- 使用列生成或Benders分解
- 使用专门的分解求解器
