### 1.1 生产计划问题

```python
#!/usr/bin/env python3
"""
生产计划问题
某工厂生产多种产品，每种产品有利润和资源消耗。
确定最优生产数量。
"""

import gurobipy as gp
from gurobipy import GRB

def solve_production_planning():
    # ===== 问题数据 =====
    products = ["P1", "P2", "P3", "P4"]
    
    # 产品利润 ($/unit)
    profit = {"P1": 30, "P2": 40, "P3": 35, "P4": 25}
    
    # 资源消耗 (unit/unit of product)
    resources = ["labor", "materials", "machine_time"]
    consumption = {
        ("P1", "labor"): 1,        ("P1", "materials"): 2,    ("P1", "machine_time"): 2,
        ("P2", "labor"): 2,        ("P2", "materials"): 1,    ("P2", "machine_time"): 3,
        ("P3", "labor"): 1.5,      ("P3", "materials"): 2,    ("P3", "machine_time"): 2,
        ("P4", "labor"): 1,        ("P4", "materials"): 3,    ("P4", "machine_time"): 1,
    }
    
    # 资源可用量
    capacity = {"labor": 100, "materials": 150, "machine_time": 120}
    
    # 产品最大生产量
    max_production = {"P1": 50, "P2": 40, "P3": 60, "P4": 30}
    
    # ===== 创建模型 =====
    model = gp.Model("production_planning")
    
    # ===== 决策变量 =====
    x = model.addVars(products, vtype=GRB.CONTINUOUS, lb=0, name="x")
    
    # ===== 目标函数 =====
    model.setObjective(
        gp.quicksum(profit[p] * x[p] for p in products),
        GRB.MAXIMIZE
    )
    
    # ===== 约束条件 =====
    # 资源约束
    for r in resources:
        model.addConstr(
            gp.quicksum(consumption[p, r] * x[p] for p in products) <= capacity[r],
            name=f"resource_{r}"
        )
    
    # 最大生产量约束
    for p in products:
        model.addConstr(x[p] <= max_production[p], name=f"max_{p}")
    
    # ===== 求解 =====
    model.optimize()
    
    # ===== 结果 =====
    print("=" * 60)
    print("生产计划问题求解结果")
    print("=" * 60)
    
    if model.status == GRB.OPTIMAL:
        print(f"\n最优利润: ${model.objVal}")
        print(f"求解时间: {model.runtime:.4f}秒")
        print("\n最优生产方案:")
        for p in products:
            print(f"  {p}: {x[p].x:.1f} 单位")
        
        print("\n资源使用情况:")
        for r in resources:
            used = sum(consumption[p, r] * x[p].x for p in products)
            print(f"  {r}: {used:.1f} / {capacity[r]} ({100*used/capacity[r]:.1f}%)")
    
    return model

if __name__ == "__main__":
    solve_production_planning()
```

### 1.2 投资组合优化

```python
#!/usr/bin/env python3
"""
投资组合优化问题
选择投资项目组合，在风险约束下最大化收益。
"""

import gurobipy as gp
from gurobipy import GRB
import random

def solve_portfolio_optimization():
    random.seed(42)
    
    # ===== 问题数据 =====
    investments = [f"I{i}" for i in range(10)]
    n_assets = len(investments)
    
    # 预期收益率 (%)
    returns = {inv: random.uniform(5, 15) for inv in investments}
    
    # 协方差矩阵 (简化为随机)
    cov = {}
    for i in investments:
        for j in investments:
            if i == j:
                cov[i, j] = random.uniform(1, 5)
            else:
                cov[i, j] = random.uniform(-1, 1)
    
    # 最大投资风险 (方差上限)
    max_risk = 10
    
    # 最大/最小投资额
    max_allocation = 0.3  # 单项不超过30%
    min_allocation = 0.05  # 最小5%
    total_budget = 1000000  # 总投资额
    
    # ===== 创建模型 =====
    model = gp.Model("portfolio_optimization")
    
    # ===== 决策变量 =====
    # x[i] = 投资于项目i的比例
    x = model.addVars(investments, vtype=GRB.CONTINUOUS, lb=0, ub=max_allocation, name="x")
    
    # ===== 目标函数 =====
    # 最大化预期收益
    model.setObjective(
        gp.quicksum(returns[i] * x[i] for i in investments),
        GRB.MAXIMIZE
    )
    
    # ===== 约束条件 =====
    # 总投资比例 = 1
    model.addConstr(gp.quicksum(x[i] for i in investments) == 1, name="budget")
    
    # 风险约束 (组合方差)
    model.addConstr(
        gp.quicksum(cov[i, j] * x[i] * x[j] for i in investments for j in investments) 
        <= max_risk,
        name="risk_constraint"
    )
    
    # 至少投资5个项目
    model.addConstr(
        gp.quicksum(x[i] >= min_allocation for i in investments) >= 5,
        name="diversification"
    )
    
    # ===== 求解 =====
    model.Params.MIPGap = 0.001
    model.optimize()
    
    # ===== 结果 =====
    print("=" * 60)
    print("投资组合优化求解结果")
    print("=" * 60)
    
    if model.status == GRB.OPTIMAL:
        print(f"\n最优预期收益率: {model.objVal:.2f}%")
        print(f"投资组合方差: {sum(cov[i,j]*x[i].x*x[j].x for i in investments for j in investments):.2f}")
        print("\n推荐投资组合:")
        selected = [(i, x[i].x) for i in investments if x[i].x > 0.01]
        selected.sort(key=lambda t: t[1], reverse=True)
        for inv, allocation in selected:
            print(f"  {inv}: {allocation*100:.1f}% (预期收益: {returns[inv]:.2f}%)")
        print(f"\n总投资金额: ${total_budget * sum(x[i].x for i in investments):,.0f}")
    
    return model

if __name__ == "__main__":
    solve_portfolio_optimization()
```

### 2.1 最大流问题

```python
#!/usr/bin/env python3
"""
最大流问题
从源点到汇点的最大可能流量。
"""

import gurobipy as gp
from gurobipy import GRB

def solve_max_flow():
    # ===== 问题数据 =====
    nodes = ["s", "v1", "v2", "v3", "v4", "t"]
    source = "s"
    sink = "t"
    
    # 边容量
    capacity = {
        ("s", "v1"): 10, ("s", "v2"): 8,
        ("v1", "v2"): 5,  ("v1", "v3"): 8,
        ("v2", "v1"): 3,  ("v2", "v4"): 6,
        ("v3", "v2"): 4,  ("v3", "t"): 9,
        ("v4", "v3"): 5,  ("v4", "t"): 7,
    }
    
    # ===== 创建模型 =====
    model = gp.Model("max_flow")
    
    # ===== 决策变量 =====
    # flow[i,j] = 从i到j的流量
    flow = model.addVars(capacity.keys(), vtype=GRB.CONTINUOUS, lb=0, name="flow")
    
    # ===== 目标函数 =====
    # 最大化汇入汇点的流量
    model.setObjective(
        gp.quicksum(flow[source, j] for i, j in capacity.keys() if i == source),
        GRB.MAXIMIZE
    )
    
    # ===== 约束条件 =====
    # 容量约束
    for (i, j), cap in capacity.items():
        model.addConstr(flow[i, j] <= cap, name=f"cap_{i}_{j}")
    
    # 流量守恒约束 (除源点和汇点外)
    for k in nodes:
        if k != source and k != sink:
            inflow = gp.quicksum(flow[i, k] for i, j in capacity.keys() if j == k)
            outflow = gp.quicksum(flow[k, j] for i, j in capacity.keys() if i == k)
            model.addConstr(inflow == outflow, name=f"flow_cons_{k}")
    
    # ===== 求解 =====
    model.optimize()
    
    # ===== 结果 =====
    print("=" * 60)
    print("最大流问题求解结果")
    print("=" * 60)
    
    if model.status == GRB.OPTIMAL:
        print(f"\n最大流量: {model.objVal}")
        print("\n流量分配:")
        for (i, j), cap in capacity.items():
            f = flow[i, j].x
            if f > 0:
                print(f"  {i} → {j}: {f} / {cap}")
    
    return model

if __name__ == "__main__":
    solve_max_flow()
```

### 2.2 最短路径问题

```python
#!/usr/bin/env python3
"""
最短路径问题
找到从源点到汇点的最短路径。
"""

import gurobipy as gp
from gurobipy import GRB

def solve_shortest_path():
    # ===== 问题数据 =====
    nodes = ["A", "B", "C", "D", "E", "F", "G"]
    
    # 边和距离
    edges = [
        ("A", "B", 4), ("A", "C", 2), ("A", "D", 5),
        ("B", "C", 1), ("B", "E", 4),
        ("C", "D", 2), ("C", "E", 3), ("C", "F", 6),
        ("D", "F", 3),
        ("E", "G", 2), ("F", "G", 2),
    ]
    
    source = "A"
    sink = "G"
    
    # ===== 创建模型 =====
    model = gp.Model("shortest_path")
    
    # ===== 决策变量 =====
    # x[i,j] = 1 如果选择边(i,j)
    x = model.addVars(
        [(e[0], e[1]) for e in edges],
        vtype=GRB.BINARY,
        name="x"
    )
    
    # ===== 目标函数 =====
    # 最小化总距离
    model.setObjective(
        gp.quicksum(dist * x[i, j] for i, j, dist in edges),
        GRB.MINIMIZE
    )
    
    # ===== 约束条件 =====
    # 流量守恒
    for k in nodes:
        if k == source:
            # 源点出度 - 入度 = 1
            model.addConstr(
                gp.quicksum(x[k, j] for i, j, _ in edges if i == k) -
                gp.quicksum(x[i, k] for i, j, _ in edges if j == k) == 1,
                name=f"source_{k}"
            )
        elif k == sink:
            # 汇点出度 - 入度 = -1
            model.addConstr(
                gp.quicksum(x[k, j] for i, j, _ in edges if i == k) -
                gp.quicksum(x[i, k] for i, j, _ in edges if j == k) == -1,
                name=f"sink_{k}"
            )
        else:
            # 中间点流量守恒
            model.addConstr(
                gp.quicksum(x[k, j] for i, j, _ in edges if i == k) ==
                gp.quicksum(x[i, k] for i, j, _ in edges if j == k),
                name=f"flow_{k}"
            )
    
    # ===== 求解 =====
    model.optimize()
    
    # ===== 结果 =====
    print("=" * 60)
    print("最短路径问题求解结果")
    print("=" * 60)
    
    if model.status == GRB.OPTIMAL:
        print(f"\n最短距离: {model.objVal}")
        
        # 重建路径
        path = [source]
        current = source
        edge_dict = {(e[0], e[1]): dist for e in edges}
        
        while current != sink:
            for i, j, d in edges:
                if i == current and x[i, j].x > 0.5:
                    path.append(j)
                    current = j
                    break
        
        print(f"\n最短路径: {' → '.join(path)}")
    
    return model

if __name__ == "__main__":
    solve_shortest_path()
```

## 3. 调度问题

### 3.1 作业车间调度 (Job Shop)

```python
#!/usr/bin/env python3
"""
作业车间调度问题 (Job Shop Scheduling)
多个作业，每个作业有多道工序，每道工序需要在指定机器上加工。
确定工序加工顺序，最小化总完工时间。
"""

import gurobipy as gp
from gurobipy import GRB

def solve_job_shop():
    # ===== 问题数据 =====
    jobs = ["J1", "J2", "J3"]
    machines = ["M1", "M2", "M3"]
    
    # 工序数据: (job, operation, machine, processing_time)
    operations = [
        # Job1: M1(3) → M2(2) → M3(4)
        ("J1", 1, "M1", 3), ("J1", 2, "M2", 2), ("J1", 3, "M3", 4),
        # Job2: M2(4) → M1(3) → M3(2)
        ("J2", 1, "M2", 4), ("J2", 2, "M1", 3), ("J2", 3, "M3", 2),
        # Job3: M1(2) → M3(3) → M2(5)
        ("J3", 1, "M1", 2), ("J3", 2, "M3", 3), ("J3", 3, "M2", 5),
    ]
    
    # ===== 创建模型 =====
    model = gp.Model("job_shop")
    
    # 参数
    H = 1000  # 时间上界
    
    # ===== 决策变量 =====
    # start[j,op] = 工序(j,op)的开始时间
    start = {}
    for j, op, m, p in operations:
        start[j, op] = model.addVar(lb=0, vtype=GRB.CONTINUOUS, name=f"start_{j}_{op}")
    
    # Cmax = 总完工时间
    Cmax = model.addVar(lb=0, vtype=GRB.CONTINUOUS, name="Cmax")
    
    # ===== 目标函数 =====
    model.setObjective(Cmax, GRB.MINIMIZE)
    
    # ===== 约束条件 =====
    # 工序顺序约束 (同一作业的工序)
    for j in jobs:
        ops = [(op, m, p) for jj, op, m, p in operations if jj == j]
        ops.sort(key=lambda x: x[0])
        for i in range(len(ops) - 1):
            op1, m1, p1 = ops[i]
            op2, m2, p2 = ops[i + 1]
            model.addConstr(
                start[j, op1] + p1 <= start[j, op2],
                name=f"seq_{j}_{op1}_{op2}"
            )
    
    # 机器冲突约束 (同一机器不能同时加工)
    for m in machines:
        ops_on_m = [(j, op, p) for j, op, mm, p in operations if mm == m]
        for i in range(len(ops_on_m)):
            for j in range(i + 1, len(ops_on_m)):
                j1, op1, p1 = ops_on_m[i]
                j2, op2, p2 = ops_on_m[j]
                
                # Big-M 约束避免同时加工
                model.addConstr(
                    start[j1, op1] + p1 <= start[j2, op2] + H * (1 - 1),
                    name=f"disj1_{j1}_{op1}_{j2}_{op2}"
                )
                model.addConstr(
                    start[j2, op2] + p2 <= start[j1, op1],
                    name=f"disj2_{j1}_{op1}_{j2}_{op2}"
                )
    
    # 总完工时间约束
    for j, op, m, p in operations:
        if op == max(o for jj, o, mm, pp in operations if jj == j):
            model.addConstr(
                start[j, op] + p <= Cmax,
                name=f"makespan_{j}"
            )
    
    # ===== 求解 =====
    model.optimize()
    
    # ===== 结果 =====
    print("=" * 60)
    print("作业车间调度问题求解结果")
    print("=" * 60)
    
    if model.status == GRB.OPTIMAL:
        print(f"\n最优总完工时间 (Makespan): {model.objVal}")
        print("\n工序开始时间:")
        for j in jobs:
            ops = [(op, m, p) for jj, op, m, p in operations if jj == j]
            ops.sort(key=lambda x: x[0])
            for op, m, p in ops:
                s = start[j, op].x
                print(f"  {j}, 工序{op} (机器{m}, 加工时间{p}): 开始={s:.1f}, 完成={s+p:.1f}")
    
    return model

if __name__ == "__main__":
    solve_job_shop()
```

## 4. 高级应用

### 4.1 两阶段随机规划

```python
#!/usr/bin/env python3
"""
两阶段随机规划示例
第一阶段: 在看到随机需求前做决策
第二阶段: 根据实现的需求调整
"""

import gurobipy as gp
from gurobipy import GRB
import random

def solve_two_stage_stochastic():
    random.seed(42)
    
    # ===== 问题数据 =====
    scenarios = ["low", "medium", "high"]
    probabilities = {"low": 0.2, "medium": 0.5, "high": 0.3}
    
    # 需求
    demand = {
        "low": {"P1": 50, "P2": 30},
        "medium": {"P1": 80, "P2": 60},
        "high": {"P1": 120, "P2": 90},
    }
    
    # 第一阶段成本 (采购成本)
    first_stage_cost = {"P1": 10, "P2": 15}
    
    # 第二阶段成本 (紧急采购成本)
    second_stage_cost = {"P1": 20, "P2": 30}
    
    # ===== 创建模型 =====
    model = gp.Model("two_stage_stochastic")
    
    # ===== 决策变量 =====
    # 第一阶段变量
    x = model.addVars(["P1", "P2"], vtype=GRB.CONTINUOUS, lb=0, name="x")
    
    # 第二阶段变量 (每个场景)
    y = {}
    for s in scenarios:
        for p in ["P1", "P2"]:
            y[s, p] = model.addVar(lb=0, vtype=GRB.CONTINUOUS, name=f"y_{s}_{p}")
    
    # ===== 目标函数 =====
    # 最小化期望总成本
    model.setObjective(
        gp.quicksum(first_stage_cost[p] * x[p] for p in ["P1", "P2"]) +
        gp.quicksum(
            probabilities[s] * second_stage_cost[p] * y[s, p]
            for s in scenarios for p in ["P1", "P2"]
        ),
        GRB.MINIMIZE
    )
    
    # ===== 约束条件 =====
    # 第二阶段需求满足约束
    for s in scenarios:
        for p in ["P1", "P2"]:
            model.addConstr(
                x[p] + y[s, p] >= demand[s][p],
                name=f"demand_{s}_{p}"
            )
    
    # ===== 求解 =====
    model.optimize()
    
    # ===== 结果 =====
    print("=" * 60)
    print("两阶段随机规划求解结果")
    print("=" * 60)
    
    if model.status == GRB.OPTIMAL:
        print(f"\n期望总成本: {model.objVal:.2f}")
        print("\n第一阶段决策 (采购量):")
        for p in ["P1", "P2"]:
            print(f"  {p}: {x[p].x:.1f} 单位")
        
        print("\n第二阶段决策 (紧急采购):")
        for s in scenarios:
            print(f"  场景 '{s}' (概率 {probabilities[s]}):")
            for p in ["P1", "P2"]:
                print(f"    {p}: {y[s, p].x:.1f} 单位")
    
    return model

if __name__ == "__main__":
    solve_two_stage_stochastic()
```
