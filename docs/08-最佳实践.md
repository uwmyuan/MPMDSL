## 1. 模型构建最佳实践

### 1.1 变量定义

#### ✅ 推荐做法

```python
# 预先计算索引列表
indices = [f"item_{i}" for i in range(n)]

# 使用有意义的变量名
production = model.addVars(products, lb=0, name="production")
demand = model.addVars(customers, lb=0, name="demand")

# 设置合理的变量边界
x = model.addVar(lb=0, ub=max_capacity, name="x")  # 避免无界变量
```

#### ❌ 避免做法

```python
# 不要使用无界变量 (可能导致无界解)
x = model.addVar(vtype=GRB.CONTINUOUS)  # 危险!

# 避免数字索引
x1, x2, x3 = model.addVar(), model.addVar(), model.addVar()

# 不要使用保留字作为变量名
model.addVar(name="for")  # 错误!
```

### 1.2 约束编写

#### ✅ 推荐做法

```python
# 分解复杂约束
raw_materials = gp.quicksum(consumption[p, r] * x[p] for p in products)
model.addConstr(raw_materials <= available[r], name="material_limit")

# 为约束命名
model.addConstr(x + y <= capacity, name="capacity_constraint")

# 使用辅助变量简化复杂表达式
z = model.addVar(name="z")
model.addConstr(z == x * y)  # 稍后线性化
```

#### ❌ 避免做法

```python
# 避免过大的约束表达式
model.addConstr(
    sum(a[i] * b[i] * c[i] * x[i] 
        for i in range(10000)) <= M  # 可能很慢
)

# 不要忽略约束命名 (调试困难)
model.addConstr(x + y <= 10)  # 无名约束
```

### 2.3 约束线性化

```python
# 二次约束线性化 (Big-M方法)
# 原: z = x * y (x, y, z 都是二进制)
# 线性化:
model.addConstr(z <= x)
model.addConstr(z <= y)
model.addConstr(z >= x + y - 1)

# 绝对值约束
# 原: |expr| <= M
# 线性化:
model.addConstr(expr <= M)
model.addConstr(-expr <= M)
```

### 2.4 对称性破缺

```python
# 箱子编号排序 (避免等价解)
for k in range(n_bins - 1):
    model.addConstr(y[k] >= y[k + 1])

# 客户分配排序
for j in range(n_facilities - 1):
    model.addConstr(
        gp.quicksum(x[i, j] for i in customers) >=
        gp.quicksum(x[i, j + 1] for i in customers)
    )

        )

        # ...
    
    def set_objective(self, profit: dict):
        """设置目标函数"""
        self._vars['profit'] = self.model.setObjective(
            gp.quicksum(profit[p] * self._vars['production'][p] 
                        for p in self._vars['production']),
            GRB.MAXIMIZE
        )
    
    def solve(self):
        """求解模型"""
        self.model.optimize()
        return self.model
    
    def get_results(self):
        """提取结果"""
        return {p: self._vars['production'][p].x 
                for p in self._vars['production']}

def validate_model(model: gp.Model) -> bool:
    """验证模型完整性"""
    
    # 检查变量数量
    num_vars = len(model.getVars())
    print(f"变量数量: {num_vars}")
    
    # 检查约束数量
    num_constrs = len(model.getConstrs())
    print(f"约束数量: {num_constrs}")
    
    # 检查是否有未绑定的变量
    for var in model.getVars():
        if var.ub > 1e9 and var.lb < -1e9:
            print(f"警告: 变量 {var.varName} 可能无界")
    
    # 检查是否有矛盾的约束
    try:
        model.optimize()
        return True
    except Exception as e:
        print(f"模型错误: {e}")
        return False
```

import time
from functools import wraps

def benchmark(func):
    """性能基准装饰器"""
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        elapsed = time.time() - start
        print(f"{func.__name__}: {elapsed:.4f}秒")
        return result
    return wrapper

@benchmark
def solve_large_instance(model):
    model.optimize()
    return model.objVal
